## Core Components

Unlike web development where you use HTML elements, React Native uses components that render to native views.

### View - The Container Component

`View` is the most fundamental component - equivalent to `<div>` in web:

```tsx
import { View, StyleSheet } from 'react-native';

export default function ContainerExample() {
  return (
    <View style={styles.container}>
      <View style={styles.box} />
      <View style={styles.box} />
      <View style={styles.box} />
    </View>
  );
}

const styles = StyleSheet.create({
  container: {
    flex: 1,
    flexDirection: 'row',
    justifyContent: 'space-around',
    alignItems: 'center',
    backgroundColor: '#f5f5f5',
  },
  box: {
    width: 80,
    height: 80,
    backgroundColor: '#3b82f6',
    borderRadius: 8,
  },
});
```

### Text - Displaying Text

All text must be wrapped in a `Text` component:

```tsx
import { Text, StyleSheet, View } from 'react-native';

export default function TextExample() {
  return (
    <View style={styles.container}>
      {/* Basic text */}
      <Text style={styles.title}>Hello, React Native!</Text>
      
      {/* Nested text for styling */}
      <Text style={styles.paragraph}>
        This is <Text style={styles.bold}>bold</Text> and this is{' '}
        <Text style={styles.italic}>italic</Text>.
      </Text>
      
      {/* Text with number of lines */}
      <Text numberOfLines={2} style={styles.truncated}>
        This is a very long text that will be truncated after two lines. 
        You won't see the rest of this text because it exceeds the limit.
      </Text>
      
      {/* Selectable text */}
      <Text selectable style={styles.selectable}>
        Long press to select this text
      </Text>
    </View>
  );
}

const styles = StyleSheet.create({
  container: {
    padding: 20,
  },
  title: {
    fontSize: 28,
    fontWeight: 'bold',
    color: '#1f2937',
    marginBottom: 16,
  },
  paragraph: {
    fontSize: 16,
    lineHeight: 24,
    color: '#4b5563',
    marginBottom: 16,
  },
  bold: {
    fontWeight: 'bold',
  },
  italic: {
    fontStyle: 'italic',
  },
  truncated: {
    fontSize: 14,
    color: '#6b7280',
    marginBottom: 16,
  },
  selectable: {
    fontSize: 14,
    color: '#3b82f6',
  },
});
```

### Image - Displaying Images

```tsx
import { Image, StyleSheet, View } from 'react-native';

export default function ImageExample() {
  return (
    <View style={styles.container}>
      {/* Local image */}
      <Image
        source={require('../assets/images/logo.png')}
        style={styles.localImage}
      />
      
      {/* Remote image (must specify dimensions) */}
      <Image
        source={{ uri: 'https://picsum.photos/200' }}
        style={styles.remoteImage}
      />
      
      {/* Image with resize modes */}
      <Image
        source={{ uri: 'https://picsum.photos/400/200' }}
        style={styles.coverImage}
        resizeMode="cover"  // cover, contain, stretch, center
      />
      
      {/* Image with loading placeholder */}
      <Image
        source={{ uri: 'https://picsum.photos/300' }}
        style={styles.placeholderImage}
        defaultSource={require('../assets/images/placeholder.png')}
      />
    </View>
  );
}

const styles = StyleSheet.create({
  container: {
    padding: 20,
    alignItems: 'center',
  },
  localImage: {
    width: 100,
    height: 100,
    marginBottom: 20,
  },
  remoteImage: {
    width: 200,
    height: 200,
    borderRadius: 100,
    marginBottom: 20,
  },
  coverImage: {
    width: '100%',
    height: 150,
    borderRadius: 8,
    marginBottom: 20,
  },
  placeholderImage: {
    width: 150,
    height: 150,
  },
});
```

<Warning>
**Remote images require explicit dimensions.** Unlike web, React Native can't automatically determine remote image sizes. Always specify `width` and `height`.
</Warning>

### TextInput - User Input

```tsx
import { TextInput, StyleSheet, View, Text, useState } from 'react-native';

export default function TextInputExample() {
  const [text, setText] = useState('');
  const [password, setPassword] = useState('');
  const [multiline, setMultiline] = useState('');

  return (
    <View style={styles.container}>
      {/* Basic text input */}
      <TextInput
        style={styles.input}
        placeholder="Enter your name"
        placeholderTextColor="#9ca3af"
        value={text}
        onChangeText={setText}
        autoCapitalize="words"
      />
      
      {/* Password input */}
      <TextInput
        style={styles.input}
        placeholder="Enter password"
        placeholderTextColor="#9ca3af"
        value={password}
        onChangeText={setPassword}
        secureTextEntry
        autoComplete="password"
      />
      
      {/* Email input */}
      <TextInput
        style={styles.input}
        placeholder="Enter email"
        placeholderTextColor="#9ca3af"
        keyboardType="email-address"
        autoCapitalize="none"
        autoComplete="email"
      />
      
      {/* Number input */}
      <TextInput
        style={styles.input}
        placeholder="Enter amount"
        placeholderTextColor="#9ca3af"
        keyboardType="numeric"
      />
      
      {/* Multiline input */}
      <TextInput
        style={[styles.input, styles.multiline]}
        placeholder="Write your message..."
        placeholderTextColor="#9ca3af"
        value={multiline}
        onChangeText={setMultiline}
        multiline
        numberOfLines={4}
        textAlignVertical="top"
      />
      
      <Text style={styles.preview}>You typed: {text}</Text>
    </View>
  );
}

const styles = StyleSheet.create({
  container: {
    padding: 20,
  },
  input: {
    borderWidth: 1,
    borderColor: '#d1d5db',
    borderRadius: 8,
    padding: 12,
    fontSize: 16,
    marginBottom: 16,
    backgroundColor: '#fff',
  },
  multiline: {
    height: 120,
    textAlignVertical: 'top',
  },
  preview: {
    fontSize: 14,
    color: '#6b7280',
    marginTop: 8,
  },
});
```

### ScrollView - Scrollable Content

```tsx
import { ScrollView, StyleSheet, View, Text } from 'react-native';

export default function ScrollViewExample() {
  const items = Array.from({ length: 20 }, (_, i) => `Item ${i + 1}`);

  return (
    <ScrollView 
      style={styles.scrollView}
      contentContainerStyle={styles.contentContainer}
      showsVerticalScrollIndicator={false}
    >
      {items.map((item, index) => (
        <View key={index} style={styles.item}>
          <Text style={styles.itemText}>{item}</Text>
        </View>
      ))}
    </ScrollView>
  );
}

// Horizontal ScrollView
export function HorizontalScrollExample() {
  return (
    <ScrollView 
      horizontal
      showsHorizontalScrollIndicator={false}
      contentContainerStyle={styles.horizontalContent}
    >
      {[1, 2, 3, 4, 5].map((num) => (
        <View key={num} style={styles.card}>
          <Text style={styles.cardText}>Card {num}</Text>
        </View>
      ))}
    </ScrollView>
  );
}

const styles = StyleSheet.create({
  scrollView: {
    flex: 1,
  },
  contentContainer: {
    padding: 20,
  },
  item: {
    backgroundColor: '#fff',
    padding: 20,
    marginBottom: 12,
    borderRadius: 8,
    shadowColor: '#000',
    shadowOffset: { width: 0, height: 2 },
    shadowOpacity: 0.1,
    shadowRadius: 4,
    elevation: 3,
  },
  itemText: {
    fontSize: 16,
    color: '#1f2937',
  },
  horizontalContent: {
    paddingHorizontal: 20,
    paddingVertical: 10,
  },
  card: {
    width: 150,
    height: 100,
    backgroundColor: '#3b82f6',
    borderRadius: 12,
    marginRight: 12,
    justifyContent: 'center',
    alignItems: 'center',
  },
  cardText: {
    color: '#fff',
    fontSize: 16,
    fontWeight: 'bold',
  },
});
```

<Note>
**Use FlatList for long lists!** ScrollView renders all children at once. For lists with many items, use `FlatList` which only renders visible items.
</Note>

## Flexbox Layout System

<Frame>
  <img src="/images/courses/react-native-crash-course/flexbox-layout.svg" alt="Flexbox Layout System" />
</Frame>

React Native uses Flexbox for layout, with some key differences from web CSS:

### Key Differences from Web

```tsx
// React Native defaults vs Web CSS defaults
const defaults = {
  flexDirection: 'column',  // Web: 'row'
  alignItems: 'stretch',    // Same as web
  flexShrink: 0,           // Web: 1
};
```

### flexDirection

Controls the primary axis:

```tsx
import { View, Text, StyleSheet } from 'react-native';

export default function FlexDirectionExample() {
  return (
    <View style={styles.container}>
      {/* Row (horizontal) */}
      <View style={styles.row}>
        <View style={[styles.box, { backgroundColor: '#ef4444' }]} />
        <View style={[styles.box, { backgroundColor: '#f59e0b' }]} />
        <View style={[styles.box, { backgroundColor: '#22c55e' }]} />
      </View>
      
      {/* Column (vertical - default) */}
      <View style={styles.column}>
        <View style={[styles.box, { backgroundColor: '#3b82f6' }]} />
        <View style={[styles.box, { backgroundColor: '#8b5cf6' }]} />
        <View style={[styles.box, { backgroundColor: '#ec4899' }]} />
      </View>
    </View>
  );
}

const styles = StyleSheet.create({
  container: {
    flex: 1,
    padding: 20,
  },
  row: {
    flexDirection: 'row',
    marginBottom: 20,
  },
  column: {
    flexDirection: 'column',
  },
  box: {
    width: 60,
    height: 60,
    margin: 5,
    borderRadius: 8,
  },
});
```

### justifyContent (Main Axis)

```tsx
// justifyContent values:
// 'flex-start' - Items at start (default)
// 'flex-end' - Items at end
// 'center' - Items centered
// 'space-between' - Equal space between items
// 'space-around' - Equal space around items
// 'space-evenly' - Equal space between and around

<View style={{ 
  flexDirection: 'row', 
  justifyContent: 'space-between',
  padding: 20 
}}>
  <View style={styles.box} />
  <View style={styles.box} />
  <View style={styles.box} />
</View>
```

### alignItems (Cross Axis)

```tsx
// alignItems values:
// 'stretch' - Stretch to fill (default)
// 'flex-start' - Items at start
// 'flex-end' - Items at end
// 'center' - Items centered
// 'baseline' - Align text baselines

<View style={{ 
  flexDirection: 'row', 
  alignItems: 'center',
  height: 200 
}}>
  <View style={{ width: 60, height: 40, backgroundColor: 'red' }} />
  <View style={{ width: 60, height: 80, backgroundColor: 'green' }} />
  <View style={{ width: 60, height: 60, backgroundColor: 'blue' }} />
</View>
```

### flex Property

```tsx
// flex: 1 means "take up remaining space"
// Multiple flex values divide space proportionally

<View style={{ flex: 1, flexDirection: 'row' }}>
  <View style={{ flex: 1, backgroundColor: 'red' }} />     {/* 25% */}
  <View style={{ flex: 2, backgroundColor: 'green' }} />   {/* 50% */}
  <View style={{ flex: 1, backgroundColor: 'blue' }} />    {/* 25% */}
</View>
```

### Complete Flexbox Example

```tsx
import { View, Text, StyleSheet } from 'react-native';

export default function FlexboxMasterExample() {
  return (
    <View style={styles.container}>
      {/* Header - fixed height */}
      <View style={styles.header}>
        <Text style={styles.headerText}>Header</Text>
      </View>
      
      {/* Main content - takes remaining space */}
      <View style={styles.main}>
        {/* Sidebar */}
        <View style={styles.sidebar}>
          <Text style={styles.sidebarText}>Sidebar</Text>
        </View>
        
        {/* Content area */}
        <View style={styles.content}>
          <View style={styles.card}>
            <Text>Card 1</Text>
          </View>
          <View style={styles.card}>
            <Text>Card 2</Text>

        </View>

      {/* Footer - fixed height */}
      <View style={styles.footer}>
        <Text style={styles.footerText}>Footer</Text>

}

const styles = StyleSheet.create({
  container: {
    flex: 1,
  },

    flex: 1,
```

## StyleSheet API

### Creating Styles

```tsx
import { StyleSheet } from 'react-native';

// Create styles using StyleSheet.create()
const styles = StyleSheet.create({
  container: {
    flex: 1,
    padding: 20,
  },
  title: {
    fontSize: 24,
    fontWeight: 'bold',
  },
});

// Benefits:
// ✓ Validation at compile time
// ✓ Performance optimization (styles created once)
// ✓ Better IDE support
```

### Combining Styles

```tsx
// Array syntax for multiple styles
<View style={[styles.container, styles.centered]} />

// Conditional styles
<View style={[
  styles.button,
  isActive && styles.buttonActive,
  isDisabled && styles.buttonDisabled,
]} />

// Inline overrides
<View style={[styles.box, { backgroundColor: 'red' }]} />
```

### Dynamic Styles

```tsx
// Using a function for dynamic styles
const getDynamicStyle = (isActive: boolean) => ({
  backgroundColor: isActive ? '#3b82f6' : '#e5e7eb',
  transform: [{ scale: isActive ? 1.05 : 1 }],
});

<View style={[styles.card, getDynamicStyle(isActive)]} />
```

## Platform-Specific Styling

### Platform Module

```tsx
import { Platform, StyleSheet } from 'react-native';

const styles = StyleSheet.create({
  container: {
    ...Platform.select({
      ios: {
        shadowColor: '#000',
        shadowOffset: { width: 0, height: 2 },
        shadowOpacity: 0.25,
        shadowRadius: 4,
      },
      android: {
        elevation: 4,
      },
      default: {
        // web and other platforms
        boxShadow: '0 2px 4px rgba(0,0,0,0.25)',
      },
    }),
  },
  
  // Platform-specific values
  padding: Platform.OS === 'ios' ? 20 : 16,
  
  // Version-specific
  paddingTop: Platform.OS === 'android' && Platform.Version >= 21 ? 24 : 0,
});
```

### Platform-Specific Files

```
// React Native automatically picks the right file:
Button.ios.tsx      // Used on iOS
Button.android.tsx  // Used on Android
Button.tsx          // Fallback for other platforms
```

## Common Styling Patterns

### Shadows

```tsx
const styles = StyleSheet.create({
  // iOS shadows
  iosShadow: {
    shadowColor: '#000',
    shadowOffset: { width: 0, height: 2 },
    shadowOpacity: 0.25,
    shadowRadius: 4,
  },
  
  // Android elevation
  androidShadow: {
    elevation: 4,
  },
  
  // Cross-platform
  shadow: {
    ...Platform.select({
      ios: {
        shadowColor: '#000',
        shadowOffset: { width: 0, height: 2 },
        shadowOpacity: 0.25,
        shadowRadius: 4,
      },
      android: {
        elevation: 4,
      },
    }),
  },
});
```

### Border Radius

```tsx
const styles = StyleSheet.create({
  // All corners
  roundedAll: {
    borderRadius: 8,
  },
  
  // Individual corners
  roundedTop: {
    borderTopLeftRadius: 8,
    borderTopRightRadius: 8,
  },
  
  // Circle (for square containers)
  circle: {
    width: 100,
    height: 100,
    borderRadius: 50,  // Half of width/height
  },
});
```

### Absolute Positioning

```tsx
const styles = StyleSheet.create({
  container: {
    position: 'relative',  // Parent must be positioned
  },
  
  absoluteChild: {
    position: 'absolute',
    top: 10,
    right: 10,
    // Can also use: bottom, left
  },
  
  // Fill parent
  overlay: {
    ...StyleSheet.absoluteFillObject,
    // Equivalent to:
    // position: 'absolute',
    // top: 0, right: 0, bottom: 0, left: 0,
  },
});
```

## Hands-On: Profile Card Component

```tsx
import { View, Text, Image, StyleSheet, Pressable } from 'react-native';
import { Ionicons } from '@expo/vector-icons';

interface ProfileCardProps {
  name: string;
  role: string;
  avatar: string;
  stats: {
    posts: number;
    followers: number;
    following: number;
  };
  onFollow?: () => void;
}

export default function ProfileCard({ 
  name, 
  role, 
  avatar, 
  stats,
  onFollow 
}: ProfileCardProps) {
  return (
    <View style={styles.card}>
      {/* Avatar */}
      <Image 
        source={{ uri: avatar }} 
        style={styles.avatar}
      />
      
      {/* User Info */}
      <View style={styles.info}>
        <Text style={styles.name}>{name}</Text>
        <Text style={styles.role}>{role}</Text>
      </View>
      
      {/* Stats */}
      <View style={styles.stats}>
        <StatItem label="Posts" value={stats.posts} />
        <StatItem label="Followers" value={stats.followers} />
        <StatItem label="Following" value={stats.following} />
      </View>
      
      {/* Follow Button */}
      <Pressable 
        style={({ pressed }) => [
          styles.button,
          pressed && styles.buttonPressed
        ]}
        onPress={onFollow}
      >
        <Ionicons name="person-add" size={16} color="#fff" />
        <Text style={styles.buttonText}>Follow</Text>
      </Pressable>
    </View>
  );
}

function StatItem({ label, value }: { label: string; value: number }) {
  return (
    <View style={styles.statItem}>
      <Text style={styles.statValue}>
        {value >= 1000 ? `${(value/1000).toFixed(1)}k` : value}
      </Text>
      <Text style={styles.statLabel}>{label}</Text>
    </View>
  );
}

const styles = StyleSheet.create({
  card: {
    backgroundColor: '#fff',
    borderRadius: 16,
    padding: 24,
    alignItems: 'center',
    ...Platform.select({
      ios: {
        shadowColor: '#000',
        shadowOffset: { width: 0, height: 4 },
        shadowOpacity: 0.1,
        shadowRadius: 12,
      },
      android: {
        elevation: 8,

    width: 100,
    height: 100,
    borderRadius: 50,
    marginBottom: 16,
    borderWidth: 3,
    borderColor: '#e5e7eb',
  },
  info: {
    alignItems: 'center',
    marginBottom: 20,
  },
  name: {
    fontSize: 24,
    fontWeight: 'bold',
    color: '#1f2937',
    marginBottom: 4,
  },
  role: {
    fontSize: 14,
    color: '#6b7280',
  },
  stats: {
    flexDirection: 'row',
    justifyContent: 'space-around',
    width: '100%',
    paddingVertical: 16,
    borderTopWidth: 1,
    borderBottomWidth: 1,
    borderColor: '#e5e7eb',
    marginBottom: 20,
  },
  statItem: {
    alignItems: 'center',
  },
  statValue: {
    fontSize: 20,
    fontWeight: 'bold',
    color: '#1f2937',
  },
  statLabel: {
    fontSize: 12,
    color: '#6b7280',
    marginTop: 2,
  },
  button: {
    flexDirection: 'row',
    alignItems: 'center',
    backgroundColor: '#3b82f6',
```
