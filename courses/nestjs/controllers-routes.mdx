# Controllers & Routing

## Route Parameters
```typescript
@Get(':id')
findOne(@Param('id') id: string) {
  return `User #${id}`;
}
```

## Request Body
```typescript
@Post()
create(@Body() createDto: CreateUserDto) {
  return this.service.create(createDto);
}
```

## Validation
```typescript
export class CreateUserDto {
  @IsString()
  name: string;
  
  @IsEmail()
  email: string;
}
```
